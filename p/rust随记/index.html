use std::io::{self, Write};
use std::thread;
use std::sync::{Arc, Mutex};

// 定义一个宏用于日志输出
macro_rules! log {
    ($($arg:tt)*) => ({
        println!("[LOG]: {}", format!($($arg)*));
    })
}

// 定义任务状态的枚举
#[derive(Debug)]
enum Status {
    Pending,
    Completed,
}

// 定义一个泛型任务结构体
struct Task<T> {
    id: usize,
    description: T,
    status: Status,
}

impl<T> Task<T> {
    fn new(id: usize, description: T) -> Self {
        Task {
            id,
            description,
            status: Status::Pending,
        }
    }

    fn complete(&mut self) {
        self.status = Status::Completed;
        log!("Task {} marked as completed.", self.id);
    }
}

// 定义命令的枚举
enum Command {
    Add(String),
    List,
    Complete(usize),
    Exit,
}

impl Command {
    // 解析用户输入为命令
    fn parse(input: &str) -> Option<Self> {
        let parts: Vec<&str> = input.trim().splitn(2, ' ').collect();
        match parts[0].to_lowercase().as_str() {
            "add" => {
                if parts.len() < 2 {
                    None
                } else {
                    Some(Command::Add(parts[1].to_string()))
                }
            },
            "list" => Some(Command::List),
            "complete" => {
                if parts.len() < 2 {
                    None
                } else {
                    parts[1].parse::<usize>().ok().map(Command::Complete)
                }
            },
            "exit" => Some(Command::Exit),
            _ => None,
        }
    }
}

// 定义待办事项应用的主要结构体
struct TodoApp {
    tasks: Arc<Mutex<Vec<Task<String>>>>,
    next_id: usize,
}

impl TodoApp {
    fn new() -> Self {
        TodoApp {
            tasks: Arc::new(Mutex::new(Vec::new())),
            next_id: 1,
        }
    }

    fn run(&mut self) {
        loop {
            print!("> ");
            io::stdout().flush().unwrap();

            let mut input = String::new();
            if io::stdin().read_line(&mut input).is_err() {
                println!("Failed to read line");
                continue;
            }

            match Command::parse(&input) {
                Some(Command::Add(desc)) => {
                    let task = Task::new(self.next_id, desc);
                    self.tasks.lock().unwrap().push(task);
                    log!("Added task {}", self.next_id);
                    self.next_id += 1;
                },
                Some(Command::List) => {
                    let tasks = self.tasks.lock().unwrap();
                    for task in tasks.iter() {
                        println!(
                            "{}: [{}] {}",
                            task.id,
                            match task.status {
                                Status::Pending => "Pending",
                                Status::Completed => "Completed",
                            },
                            task.description
                        );
                    }
                },
                Some(Command::Complete(id)) => {
                    let tasks = self.tasks.lock().unwrap();
                    if let Some(task) = tasks.iter_mut().find(|t| t.id == id) {
                        task.complete();
                    } else {
                        println!("Task with ID {} not found.", id);
                    }
                },
                Some(Command::Exit) => {
                    println!("Exiting...");
                    break;
                },
                None => {
                    println!("Invalid command.");
                },
            }
        }
    }
}

fn main() {
    let mut app = TodoApp::new();

    // 使用线程来运行应用，这里仅为示例展示并发
    let app_handle = thread::spawn(move || {
        app.run();
    });

    // 等待线程完成
    app_handle.join().unwrap();
}

宏

Rust 的宏（Macros）是其强大元编程能力的重要组成部分，允许开发者在编译时生成代码，从而实现代码的复用和简化复杂的模式。理解宏的语法和使用方法对于深入掌握 Rust 编程非常关键。

宏的基本概念

在 Rust 中，宏是一种在编译时展开的代码生成工具。它们与函数不同，函数在运行时执行，而宏在编译阶段进行代码替换。Rust 提供了两种主要类型的宏：

  1. 声明宏（Declarative Macros）：也称为 macro_rules! 宏，通过模式匹配进行代码生成。
  2. 过程宏（Procedural Macros）：包括自定义派生宏（Derive Macros）、属性宏（Attribute Macros）和函数宏（Function-like Macros），它们通过编写函数来处理输入的代码并生成新的代码。

声明宏（macro_rules!）

声明宏是 Rust 最早和最常用的宏类型，使用 macro_rules! 关键字定义。它们通过匹配特定的模式来展开代码，允许在多个地方重用相似的代码模式。

macro_rules! 宏名称 {
    (模式) => {
        展开的代码
    };
    // 可以定义多个模式匹配
}
  • 模式：类似于函数的参数，定义了宏接受的输入形式。
  • 展开的代码：根据匹配到的模式生成的代码，可以包含变量绑定、重复模式等。

常用的宏片段

  • 标识符（ident）：匹配变量名、函数名等标识符。
  • 表达式（expr）：匹配任意表达式。

宏的调试与错误处理

由于宏在编译时展开，调试宏可能会比较复杂。Rust 提供了一些工具和技巧来帮助调试宏，例如使用 println! 在宏中输出调试信息，或者利用编译器的错误信息来定位问题。

此外，宏的错误信息有时可能不够直观，因此编写宏时应尽量保持模式简单，必要时分解成多个小宏，提高可读性和可维护性。

过程宏（Procedural Macros）

过程宏是一种更强大的宏类型，允许开发者以函数的形式处理输入的代码并生成新的代码。过程宏主要分为以下几类：

  1. 自定义派生宏（Derive Macros）：用于自动为结构体或枚举生成实现特定特征的代码。例如，常用的 #[derive(Debug)]。
  2. 属性宏（Attribute Macros）：允许开发者为项添加自定义属性，并基于这些属性生成代码。
  3. 函数宏（Function-like Macros）：类似于声明宏，但语法上更像函数调用，允许更复杂的代码生成逻辑。

use proc_macro::TokenStream;
use quote::quote;
use syn;

#[proc_macro_derive(HelloMacro)]
pub fn hello_macro_derive(input: TokenStream) -> TokenStream {
    // 解析输入的语法树
    let ast = syn::parse(input).unwrap();
    // 生成代码
    impl_hello_macro(&ast)
}

fn impl_hello_macro(ast: &syn::DeriveInput) -> TokenStream {
    let name = &ast.ident;
    let gen = quote! {
        impl HelloMacro for #name {
            fn hello() {
                println!("Hello, Macro! My name is {}", stringify!(#name));
            }
        }
    };
    gen.into()
}

宏的最佳实践

  1. 保持宏简单：复杂的宏难以阅读和维护，尽量保持宏逻辑简单，必要时分解成多个小宏。
  2. 使用明确的模式：清晰地定义宏的输入模式，避免模糊或过于宽泛的匹配，减少意外的代码生成。
  3. 提供良好的文档：为宏编写详细的文档和示例，帮助使用者理解其用法和限制。
  4. 避免过度使用：虽然宏强大，但过度使用可能导致代码难以理解。优先考虑使用函数、泛型和其他语言特性来实现功能，只有在必要时使用宏。

Rust 的宏系统提供了强大的代码生成和复用能力，通过声明宏和过程宏，开发者可以在编译时动态生成代码，简化复杂的模式和逻辑。理解宏的语法和使用方法需要一定的实践，但掌握宏将大大提升 Rust 编程的灵活性和效率。

枚举与模式匹配

枚举（Enum）和模式匹配（Pattern Matching）是 Rust 中非常强大且核心的语法特性，能够帮助开发者编写简洁且高效的代码。下面将详细介绍这两个概念的语法及其应用。

枚举（Enum）

枚举类型允许你定义一个类型，该类型可以是几个不同变体中的一个。与其他语言中的枚举不同，Rust 的枚举可以携带数据，这使得它们在表达复杂数据结构时非常灵活。

定义枚举：在 Rust 中，使用 enum 关键字来定义枚举。每个变体可以是一个简单的标识符，也可以包含与之关联的数据。

    Quit,
    Move { x: i32, y: i32 },

    let msg1 = Message::Quit;

fn process_message(msg: Message) {
    match msg {
        Message::Quit => println!("Quit message received."),

所有权与借用

所有权与借用（Ownership and Borrowing）是 Rust 语言中最核心且最具特色的概念之一，它们共同构成了 Rust 的内存安全性基础。

所有权（Ownership）

在 Rust 中，所有权系统负责管理内存，确保内存安全而无需垃圾回收器。每一个值在 Rust 中都有一个所有者（owner），并且在任何时刻，一个值只能有一个所有者。所有权规则如下：

  1. 每个值都有一个所有者。
  2. 每个值在同一时间只能有一个所有者。
  3. 当所有者超出作用域时，值会被自动丢弃（drop），从而释放内存。

示例：所有权的转移（Move）

use std::sync::{Arc, Mutex};
use std::thread;

fn main() {
    let counter = Arc::new(Mutex::new(0)); // 使用 Arc 和 Mutex 共享数据

生命周期

生命周期（Lifetimes）是 Rust 中一个关键但较为抽象的概念，它用于确保引用的有效性，防止悬垂引用（dangling references）和数据竞争（data races）。

生命周期的基本概念

在 Rust 中，每个引用都有一个生命周期，生命周期描述了引用在内存中的有效范围。生命周期确保在引用有效期间，被引用的数据不会被销毁，从而避免悬垂引用。Rust 编译器通过生命周期检查来验证代码的安全性，确保引用的使用不会导致内存安全问题。

生命周期注解的语法

生命周期注解使用撇号（'）加一个小写字母来表示，例如 'a。生命周期注解本身并不影响运行时行为，它们只是帮助编译器理解引用之间的关系。

生命周期省略规则（Lifetime Elision）

在许多情况下，Rust 编译器可以通过生命周期省略规则（Lifetime Elision Rules）自动推断引用的生命周期，无需显式标注。这些规则主要适用于简单的函数签名。

生命周期省略规则：

  1. 每个引用参数都有一个独立的生命周期参数。
  2. 如果只有一个输入生命周期参数，那个生命周期被赋给所有输出生命周期参数。
  3. 如果有多个输入生命周期参数，但其中一个是 &self 或 &mut self，那么 self 的生命周期被赋给所有输出生命周期参数。

错误处理

错误处理是编程中不可或缺的一部分，Rust 通过其独特的错误处理机制，旨在在编译时捕捉潜在的问题，从而确保程序的可靠性和稳定性。Rust 提供了两种主要的错误处理类型：可恢复错误（Recoverable Errors） 和 不可恢复错误（Unrecoverable Errors）。

可恢复错误与不可恢复错误

在 Rust 中，错误处理分为两类：

  1. 可恢复错误（Recoverable Errors）：这些错误通常是由于外部因素引起的，如文件未找到、网络请求失败等。这类错误可以通过适当的处理机制进行恢复或提供替代方案。Rust 使用 Result 枚举来表示可恢复错误。

  2. 不可恢复错误（Unrecoverable Errors）：这些错误通常是程序内部的逻辑错误，如数组越界访问、空指针引用等。这类错误通常无法通过程序恢复，因此 Rust 使用 panic! 宏来处理不可恢复错误。

使用 Result 处理可恢复错误

Result 是一个泛型枚举，定义如下：

使用 panic! 的注意事项

虽然 panic! 提供了一种简单的方式来处理严重错误，但在实际应用中应尽量避免频繁使用，因为它会导致程序崩溃。对于可预见的错误，优先使用 Result 类型进行处理。

使用第三方错误处理库

虽然 Rust 标准库提供了强大的错误处理机制，但在复杂项目中，使用第三方库可以进一步简化错误处理和增强功能。常用的错误处理库包括 anyhow 和 thiserror。

use anyhow::{Result, Context};
use std::fs::File;
use std::io::Read;

fn read_username_from_file() -> Result<String> {
    let mut file = File::open("username.txt")
        .with_context(|| "Failed to open username.txt")?;
    let mut username = String::new();

use thiserror::Error;
use std::fs::File;
use std::io::{self, Read};

#[derive(Error, Debug)]
enum MyError {
    #[error("IO error occurred: {0}")]
    Io(#[from] io::Error),
    #[error("Parse error occurred: {0}")]
    Parse(#[from] std::num::ParseIntError),
}

fn read_number_from_file() -> Result<i32, MyError> {
    let mut file = File::open("number.txt")?;
    let mut contents = String::new();
    file.read_to_string(&mut contents)?;
    let number: i32 = contents.trim().parse()?;
    Ok(number)
}

fn main() -> Result<(), MyError> {
    let number = read_number_from_file()?;
    println!("Number: {}", number);
    Ok(())
}

错误处理最佳实践

  1. 优先使用 Result 类型处理可恢复错误：对于可能发生的错误，应优先使用 Result 类型进行显式的错误处理，而不是依赖于 panic!。

  2. 利用 ? 操作符简化错误传播：在函数链中，使用 ? 操作符可以简化错误处理逻辑，使代码更为简洁和易读。

  3. 提供上下文信息：在错误传播过程中，尽可能提供有意义的上下文信息，有助于调试和理解错误发生的原因。

  4. 定义自定义错误类型：对于复杂项目，定义自己的错误类型可以更准确地描述错误，并与 Rust 的错误处理生态系统集成。

  5. 避免不必要的 unwrap 和 expect：除非在明确知道不会出错的情况下，否则应避免使用 unwrap 和 expect，以防止程序因意外错误而崩溃。

  6. 使用第三方错误处理库：在复杂的错误处理场景中，考虑使用 anyhow 和 thiserror 等第三方库，以简化错误处理流程并增强功能。

Rust 的错误处理机制通过 Result 和 Option 类型，以及 panic! 宏，提供了灵活而强大的工具来管理程序中的各种错误。Result 类型适用于可恢复错误，允许开发者显式地处理和传播错误信息；而 panic! 适用于不可恢复错误，迅速终止程序以避免潜在的内存安全问题。通过 ? 操作符和第三方库如 anyhow 和 thiserror，Rust 提供了简洁且高效的错误处理方式，使得编写健壮的 Rust 程序变得更加容易。
